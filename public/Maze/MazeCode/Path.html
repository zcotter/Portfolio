 5 /**
 6  * This class represents the path in a maze.
 7  * A path is any space that can be moved to besides the StartPoint and EndPoint.
 8  */
 9 public class Path extends Space {
10     private boolean visited;
11 
12     /**
13      * Creates a new Path
14      * @param xLoc x coordinate of path
15      * @param yLoc y coordinate of path
16      */
17     public Path(int xLoc, int yLoc) {
18         super(xLoc, yLoc);
19         visited = false;
20     }
21 
22     /**
23      * Marks that this space has been passed over by the navigator.
24      */
25     public void visit() {
26         visited = true;
27     }
28 
29     /**
30      * Determines whether or not this space has been passed over by the navigator.
31      * @return
32      */
33     public boolean isVisited() {
34         return visited;
35     }
36 }
