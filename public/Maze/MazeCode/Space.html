 5 /**
 6  * This class represents a space within a maze represented by a grid.
 7  * Extensions of this class represent more specific types of spaces.
 8  */
 9 public abstract class Space {
10     protected int x;
11     protected int y;
12 
13     /**
14      * Creates a new Space
15      * @param xLoc x coordinate of space
16      * @param yLoc y coordinate of space
17      */
18     public Space(int xLoc, int yLoc) {
19         x = xLoc;
20         y = yLoc;
21     }
22 
23     /**
24      * Accesses the x coordinate of this space.
25      * @return integer value of x coordinate
26      */
27     public int getX() {
28         return x;
29     }
30 
31     /**
32      * Accesses the y coordinate of this space.
33      * @return integer value of y coordinate.
34      */
35     public int getY() {
36         return y;
37     }
38 }
