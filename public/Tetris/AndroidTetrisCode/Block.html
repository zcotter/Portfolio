  8 /**
  9  * Represents a Tetris block by position in the grid and color. Includes functions

 13 public class Block {

 15     private int x;

 19     /**
 20      * Constructs a test block

 23         x = -1;

 28     /**
 29      * Constructs a block using the given parameters

 34     public Block(int x,
 35                  int y,
 36                  int color) {

 38         this.y = y;

 90     /**
 91      * Moves the block down a row if the result is inbounds
 92      * @return whether or not the piece was moved
 93      */
 94     public boolean moveDown() {
 95         if (TetrisView.inbounds(x,
 96                                 y + 1)) {
 97             y += 1;
 98             return true;
 99         }
100         return false;
101     }
102 
103     /**
104      * Moves the block right a column if the result is inbounds
105      * @return whether or not the piece was moved.
106      */
107     public boolean moveRight() {
108         if (TetrisView.inbounds(x + 1,
109                                 y)) {
110             x += 1;
111             return true;
112         }
113         return false;
114     }
115 
116     /**
117      * Moves the block left a column if the result is inbounds
118      * @return whether or not the piece was moved.
119      */
120     public boolean moveLeft() {
121         if (TetrisView.inbounds(x - 1,
122                                 y)) {
123             x -= 1;
124             return true;
125         }
126         return false;
127     }
128 
129     /**
130      * Recursively calls moveDown() until it returns false.
131      */
132     public void moveDownToLowestPossible() {
133         boolean keepGoing = false;
134         while (keepGoing) {
135             keepGoing = moveDown();
136         }
137     }

139     /**
140      * determines if this block is inbounds
141      * @return Whether or not this block is currently inbounds

143     public boolean inbounds() {
144         return TetrisView.inbounds(x,
145                                    y);

148     /**
149      * Rotates this block counter clockwise around the given grid space, if the
150      * result is inbounds
151      * @param centerPoint Coordinate of axis of rotation
152      * @return whether or not the rotation was successful
153      */
154     public boolean rotateCCW(Coordinate centerPoint) {
155         int targetX = centerPoint.getX() + (centerPoint.getY() - this.getY());
156         int targetY = centerPoint.getY() + (this.getX() - centerPoint.getX());
157         if (TetrisView.inbounds(targetX,
158                                 targetY)) {
159             x = targetX;
160             y = targetY;
161             return true;
162         }
163         return false;
164     }
165 
166     /**
167      * Determines if this block can rotate counter clockwise around the given
168      * grid space
169      * @param centerPoint Coordinate of axis of rotation
170      * @return whether or not the rotation would be successful
171      */
172     public boolean canRotateCCW(Coordinate centerPoint) {
173         int targetX = centerPoint.getX() + (centerPoint.getY() - this.getY());
174         int targetY = centerPoint.getY() + (this.getX() - centerPoint.getX());
175         if (TetrisView.inbounds(targetX,
176                                 targetY)) {
177             return true;
178         }
179         return false;
180     }
181 
182     /**
183      * Rotates this block clockwise around the given grid space, if the
184      * result is inbounds
185      * @param centerPoint Coordinate of axis of rotation
186      * @return whether or not the rotation was successful
187      */
188     public boolean rotateCW(Coordinate centerPoint) {
189         final int oldX = getX();
190         final int oldY = getY();
191         boolean keepTurning = rotateCCW(centerPoint);
192         if (keepTurning) {
193             keepTurning = rotateCCW(centerPoint);
194         }
195         if (keepTurning) {
196             keepTurning = rotateCCW(centerPoint);
197         }
198         if (!keepTurning) {
199             x = oldX;
200             y = oldY;
201             return false;
202         }
203         else {
204             return true;
205         }
206     }
207 
208     /**
209      * Determines if this block can rotate clockwise around the given
210      * grid space
211      * @param centerPoint Coordinate of axis of rotation
212      * @return whether or not the rotation would be successful
213      */
214     public boolean canRotateCW(Coordinate centerPoint) {
215         Block testBlock = new Block(this.getX(),
216                                     this.getY(),
217                                     this.getColor());
218         if (testBlock.canRotateCCW(centerPoint)) {
219             testBlock.rotateCCW(centerPoint);
220             if (testBlock.canRotateCCW(centerPoint)) {
221                 testBlock.rotateCCW(centerPoint);
222                 if (testBlock.canRotateCCW(centerPoint)) {
223                     return true;
224                 }
225             }
226         }
227         return false;
228     }

281     void paint(Canvas canvas) {
282         int left = x * TetrisView.blockWidth;
283         int right = left + TetrisView.blockWidth;
284         int top = y * TetrisView.blockHeight;
285         int bottom = top + TetrisView.blockHeight;
286         canvas.drawRect(new Rect(left,
287                                  top,
288                                  right,
289                                  bottom),
290                         getPaint());
291     }
