146     public Tetra(char identifier) {
147         if (identifier == 'o') {
148             constructorHelper(O_BLOCKS,
149                               O_CENTER);
150         }
151         else if (identifier == 'i') {
152             constructorHelper(I_BLOCKS,
153                               I_CENTER);
154         }

156             constructorHelper(L_BLOCKS,
157                               L_CENTER);
158         }
159         else if (identifier == 'j') {
160             constructorHelper(J_BLOCKS,
161                               J_CENTER);
162         }
163         else if (identifier == 't') {
164             constructorHelper(T_BLOCKS,
165                               T_CENTER);
166         }
167         else if (identifier == 'z') {
168             constructorHelper(Z_BLOCKS,
169                               Z_CENTER);
170         }
171         else if (identifier == 's') {
172             constructorHelper(S_BLOCKS,
173                               S_CENTER);
174         }
175     }

199     private void constructorHelper(Block[] blocks,
200                                    Coordinate center) {
201         tetra = new HashSet<Block>();
202         for (Block b : blocks) {
203             tetra.add(b);
204         }
205         this.center = center;
206     }

303     /**
304      * Moves all of the blocks in this tetra down if and only if they can all
305      * be moved down.
306      * @return Whether or not a movement could be made.
307      */
308     public boolean moveDown() {
309         HashSet<Block> newBlocks = new HashSet<Block>();
310         for (Block b : tetra) {
311             boolean success = b.moveDown();
312             if (success) {
313                 newBlocks.add(b);
314             }
315             else {
316                 return false;
317             }
318         }
319         this.setTetra(newBlocks);
320         this.setCenter(new Coordinate(
321                 this.getCenter().getX(),
322                 this.getCenter().getY() + 1));
323         return true;
324     }

326     /**
327      * Moves all of the blocks in this tetra left if and only if they can all
328      * be moved left.
329      * @return Whether or not a movement could be made.
330      */
331     public boolean moveLeft() {
332         Tetra newTetra = new Tetra();
333         Block[] temp = tetra.toArray(new Block[1]);
334         for (int count = 0; count < temp.length; count++) {
335             Block b = temp[count];
336             boolean success = b.moveLeft();
337             if (success) {
338                 newTetra.addBlock(b);
339             }
340             else {
341                 return false;
342             }
343         }
344         this.setTetra(null);
345         this.setTetra(newTetra.getTetra());
346         this.setCenter(new Coordinate(
347                 this.getCenter().getX() - 1,
348                 this.getCenter().getY()));
349         return true;
350     }

404     /**
405      * Rotates each Block owned by this Tetra clockwise around the Tetra's
406      * center point, if and only if all rotations would be successful.
407      */
408     public void rotateClockwise() {
409         boolean possible = true;
410         for (Block b : tetra) {
411             if (b.canRotateCW(center) == false) {
412                 possible = false;
413                 break;
414             }
415         }
416         if (possible) {
417             for (Block b : tetra) {
418                 b.rotateCW(center);
419             }
420         }
421     }
