 1 /**
 2  * Represents integer cartesian coordinates.
 3  * @author Zach Cotter
 4  */
 5 public class Coordinate{
 6     private int x;
 7     private int y;
 8 
 9     /**
10      * Creates a new coordinate (-1,-1).
11      */
12     public Coordinate() {
13         x = -1;
14         y = -1;
15     }
16 
17     /**
18      * Creates a new coordinate (x,y)
19      * @param x int representing x value of new coordinate
20      * @param y int representing y value of new coordinate
21      */
22     public Coordinate(int x, int y) {
23         this.x = x;
24         this.y = y;
25     }
26 
27     /**
28      * Accessor for x value
29      * @return int representing x value of this
30      */
31     public int getX() {
32         return x;
33     }
34 
35     /**
36      * Accessor for y value
37      * @return int representing y value of this
38      */
39     public int getY() {
40         return y;
41     }
42 
43     /**
44      * Mutator for x value
45      * @param x int representing new x value of this
46      */
47     public void setX(int x) {
48         this.x = x;
49     }
50 
51     /**
52      * Mutator for y value
53      * @param y int representing new y value of this
54      */
55     public void setY(int y) {
56         this.y = y;
57     }
58 
59     /**
60      * Returns a String representing this.
61      * @return A String representation of this coordinate: (x,y)
62      */
63     @Override
64     public String toString() {
65         return "(" + getX() + ", " + getY() + ")";
66     }
67 
68     /**
69      * Determines if this is equivalent to the given object, if it is even an
70      * instanceof Coordinate
71      * @param obj Object to be compared to this.
72      * @return Whether or not the objects are equivalent
73      */
74     @Override
75     public boolean equals(Object obj) {
76         if(obj instanceof Coordinate){
77             Coordinate o = (Coordinate) obj;
78             return getX() == o.getX() && getY() == o.getY();
79         }
80         return false;
81     }
82 }
