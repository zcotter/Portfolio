 65     /**
 66      * Constructs a tetra specific to Tetris. Tetras are identified by a
 67      * letter that appears similar to the grouping of their Blocks.
 68      * @param identifier char representing predefined target Tetra
 69      */
 70     public Tetra(char identifier) {
 71         if (identifier == 'o') {
 72             constructorHelper(O_BLOCKS, O_CENTER);
 73         } else if (identifier == 'i') {
 74             constructorHelper(I_BLOCKS, I_CENTER);
 75         } else if (identifier == 'l') {
 76             constructorHelper(L_BLOCKS, L_CENTER);
 77         } else if (identifier == 'j'){
 78             constructorHelper(J_BLOCKS, J_CENTER);
 79         } else if (identifier == 't'){
 80             constructorHelper(T_BLOCKS, T_CENTER);
 81         } else if (identifier == 'z'){
 82             constructorHelper(Z_BLOCKS, Z_CENTER);
 83         } else if (identifier == 's'){
 84             constructorHelper(S_BLOCKS, S_CENTER);
 85         }
 86     }

107     private void constructorHelper(Block[] blocks, Coordinate center) {
108         tetra = new HashSet<Block>();
109         tetra.addAll(Arrays.asList(blocks));
110         this.center = center;

113     /**
114      * Instantiates a new Tetra with the same values as this. Calls copy
115      * constructors of each field with as much depth as necessary to make
116      * the new Tetra completely independent.
117      * @return a new, identical Tetra
118      */
119     public Tetra cloneMethod(){
120         HashSet<Block> cloneSet = new HashSet<Block>();
121         for (Block b : tetra) {
122             int x = b.getX();
123             int y = b.getY();
124             Color color = b.getColor();
125             int r = color.getRed();
126             int g = color.getGreen();
127             int bl = color.getBlue();
128             cloneSet.add(new Block(x, y, new Color(r,g,bl)));
129         }
130         return new Tetra(cloneSet, new Coordinate(1000, 1000));
131     }

226     /**
227      * Moves all of the blocks in this tetra down if and only if they can all
228      * be moved down.
229      * @return Whether or not a movement could be made.
230      */
231     public boolean moveDown() {
232         HashSet<Block> newBlocks = new HashSet<Block>();
233         for (Block b : tetra) {
234             boolean success = b.moveDown();
235             if (success) {
236                 newBlocks.add(b);
237             } else {
238                 return false;
239             }
240         }
241         this.setTetra(newBlocks);
242         this.setCenter(new Coordinate(
243                 this.getCenter().getX(),
244                 this.getCenter().getY() + 1));
245         return true;
246     }

248     /**
249      * Moves all of the blocks in this tetra left if and only if they can all
250      * be moved left.
251      * @return Whether or not a movement could be made.
252      */
253     public boolean moveLeft() {
254         Tetra newTetra = new Tetra();
255         Block[] temp = tetra.toArray(new Block[1]);
256         for (int count = 0; count < temp.length; count++) {
257             Block b = temp[count];
258             boolean success = b.moveLeft();
259             if (success) {
260                 newTetra.addBlock(b);
261             } else {
262                 return false;
263             }
264         }
265         this.setTetra(null);
266         this.setTetra(newTetra.getTetra());
267         this.setCenter(new Coordinate(
268                        this.getCenter().getX() - 1,
269                        this.getCenter().getY()));
270         return true;
271     }

295     /**
296      * Determines if all of the blocks in the given Tetra are equivalent (for
297      * this games purposes) to this. Equivalent does not necessarily mean they
298      * are identical(reference the same memory location)
299      * @param other The Tetra to compare this to.
300      * @return True if the two Tetras are equivalent.
301      */
302     public boolean equals(Tetra other) {
303         boolean equivalent = true;
304         if (other.getTetra().size() != getTetra().size()) {
305             return false;
306         }
307         for (Block b : tetra) {
308             boolean found = false;
309             for (Block otherBlock : tetra) {
310                 if (b.equals(otherBlock)) {
311                     found = true;
312                     break;
313                 }
314             }
315             if (!found) {
316                 equivalent = false;
317                 break;
318             }
319         }
320         return equivalent;
321     }

325     /**
326      * Rotates each Block owned by this Tetra clockwise around the Tetra's
327      * center point, if and only if all rotations would be successful.
328      */
329     public void rotateClockwise() {
330         boolean possible = true;
331         for(Block b : tetra){
332             if(b.canRotateCW(center) == false){
333                 possible = false;
334                 break;
335             }
336         }
337         if(possible){
338             for(Block b : tetra){
339                 b.rotateCW(center);
340             }
341         }
342     }
